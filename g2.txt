Program            -> FunctionDef

FunctionDef        -> Type main ( ) Block

Type               -> int
                    | char
                    | float
                    | double
                    | void

Block              -> { StmtListOpt }

StmtListOpt        -> StmtList
                    | ε

StmtList           -> StmtList Stmt
                    | Stmt

Stmt               -> DeclStmt
                    | AssignStmt
                    | IfStmt
                    | WhileStmt
                    | ForStmt
                    | PostfixStmt
                    | CallStmt
                    | ReturnStmt
                    | Block

DeclStmt           -> Type InitDeclList ;

InitDeclList       -> InitDeclList , InitDecl
                    | InitDecl

InitDecl           -> id
                    | id = Expr

AssignStmt         -> LValue = Expr ;

LValue             -> id

IfStmt             -> if ( Condition ) Block ElseOpt

ElseOpt            -> else Block
                    | ε

WhileStmt          -> while ( Condition ) Block

ForStmt            -> for ( ForInitOpt ; ConditionOpt ; ForUpdateOpt ) Block

ForInitOpt         -> AssignExpr
                    | ε

ConditionOpt       -> Condition
                    | ε

ForUpdateOpt       -> UpdateExpr
                    | ε

UpdateExpr         -> id ++
                    | id --
                    | AssignExpr

PostfixStmt        -> id ++ ;
                    | id -- ;

CallStmt           -> id ( ArgListOpt ) ;

ArgListOpt         -> ArgList
                    | ε

ArgList            -> ArgList , Expr
                    | Expr

ReturnStmt         -> return Expr ;

Condition          -> Expr RelOp Expr
                    | Expr

RelOp              -> >
                    | <
                    | >=
                    | <=
                    | ==
                    | !=

AssignExpr         -> id = Expr

Expr               -> Expr AddOp Term
                    | Term

AddOp              -> +
                    | -

Term               -> Term MulOp Unary
                    | Unary

MulOp              -> *
                    | /
                    | %

Unary              -> UnaryOp Unary
                    | PostfixExpr

UnaryOp            -> +
                    | -
                    | !
                    | &
                    | *
                    | ~

PostfixExpr        -> id ++
                    | id --
                    | Primary

Primary            -> id
                    | constant
                    | literal
                    | ( Expr )
